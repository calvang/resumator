% Custom commands
\newcommand{\resumeItem}[2]{
	\item\small{
		\textbf{#1}{: #2 \vspace{-2pt}}
	}
}

\newcommand{\resumeSubheading}[4]{
	\vspace{-1pt}\item
	\begin{tabular*}{0.97\textwidth}{l@{\extracolsep{\fill}}r}
		\textbf{#1} & #2 \\
		\textit{\small#3} & \textit{\small #4} \\
	\end{tabular*}\vspace{-5pt}
}

\newcommand{\resumeSubItem}[2]{\resumeItem{#1}{#2}\vspace{-4pt}}

\renewcommand{\labelitemii}{$\circ$}

\newcommand{\resumeSubHeadingListStart}{\begin{itemize}[leftmargin=*]}
	\newcommand{\resumeSubHeadingListEnd}{\end{itemize}}
\newcommand{\resumeItemListStart}{\begin{itemize}}
	\newcommand{\resumeItemListEnd}{\end{itemize}\vspace{-5pt}}

	\BLOCK{ for section in sections }
		\section{\VAR{section.title}}
		\resumeSubHeadingListStart
		\BLOCK{ for entry in section.entries }

			\BLOCK{ if entry.subtitle }
				\resumeSubheading
				{\VAR{entry.title}}{\VAR{entry.location}}
				{\VAR{entry.subtitle}}{\VAR{entry.dates}}}
				\resumeItemListStart
				\BLOCK{ if entry.info }
					\BLOCK{ for data in entry.info }
						\resumeItem{}
						{\VAR{data}}
					\BLOCK{ endfor }
				\BLOCK{ endif }
				\resumeItemListEnd

			\BLOCK{ else }
				\item
				\textbf{\VAR{entry.title}}
				\hfill
				\textit{\VAR{entry.dates}}
				\resumeItemListStart
				\BLOCK{ if entry.info }
					\BLOCK{ for data in entry.info }
						\resumeItem{}
						{\VAR{data}}
					\BLOCK{ endfor }
				\BLOCK{ endif }
				\resumeItemListEnd
			
			\BLOCK{ endif }

		\BLOCK{ endfor }
		\resumeSubHeadingListEnd
	\BLOCK{ endfor }
